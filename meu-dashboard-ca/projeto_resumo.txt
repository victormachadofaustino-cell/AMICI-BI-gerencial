--- ARQUIVO: supabase\functions\sync-contaazul\index.ts ---
import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from "https://esm.sh/@supabase/supabase-js@2"

serve(async (req) => {
  const clientID = "579jvtneqa75470u0v5rr1s70h";
  const clientSecret = "1ld33o0q6a3tptcei2hid4circ41clmt3u7khc7clgmda1n4u79k";
  
  // 1. Pegue um novo 'code' aqui AGORA: 
  // https://api.contaazul.com/auth/authorize?redirect_uri=http://localhost:5173&client_id=579jvtneqa75470u0v5rr1s70h&scope=sales
  const authCode = "COLE_O_NOVO_CODE_AQUI"; 

  try {
    // Passo 1: Trocar o código pelo Access Token
    const authHeader = btoa(`${clientID}:${clientSecret}`);
    const tokenResp = await fetch("https://api.contaazul.com/oauth2/token", {
      method: "POST",
      headers: {
        "Authorization": `Basic ${authHeader}`,
        "Content-Type": "application/x-www-form-urlencoded"
      },
      body: new URLSearchParams({
        "grant_type": "authorization_code",
        "redirect_uri": "http://localhost:5173",
        "code": authCode
      })
    });

    const tokenData = await tokenResp.json();

    // Passo 2: Usar o token para perguntar "Quem é o dono desta conta?"
    const ownerResp = await fetch("https://api.contaazul.com/v1/owner", {
      headers: { "Authorization": `Bearer ${tokenData.access_token}` }
    });
    
    const ownerData = await ownerResp.json();

    return new Response(JSON.stringify({
      status: "Sucesso!",
      empresa_conectada: ownerData.name,
      email_cliente: ownerData.email
    }), { headers: { "Content-Type": "application/json" } });

  } catch (err) {
    return new Response(JSON.stringify({ erro: "Código expirado ou inválido. Gere um novo!" }), { status: 400 });
  }
})
